Object Methods 
let user = {
                        name: "John",
                        age: 30
                      };
                      
                      user.sayHi = function() {
                        alert("Hello!");
                      };
                      
                      user.sayHi(); // Hello!
Method Shorthand
// these objects do the same

                                user = {
                                  sayHi: function() {
                                    alert("Hello");
                                  }
                                };
                                
                                // method shorthand looks better, right?
                                user = {
                                  sayHi() { // same as "sayHi: function(){...}"
                                    alert("Hello");
                                  }
                                };
“this” in methods
let user = {
                        name: "John",
                        age: 30,
                      
                        sayHi() {
                          // "this" is the "current object"
                          alert(this.name);
                        }
                      
                      };
                      
                      user.sayHi(); // John

Finding all the Properties of an Object
for(const key in superman) {
console.log(key + ": " + superman[key]);
}

To make sure all properties are returned.


for(const key in superman) {
if(superman.hasOwnProperty(key)){
console.log(key + ": " + superman[key]);
} }

To return the array of keys, use the following code:


for(const key of Object.keys(superman)) {
console.log(key);
}
